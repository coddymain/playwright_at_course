import re
from playwright.sync_api import Playwright, sync_playwright, expect


def run(playwright: Playwright) -> None:
    browser = playwright.chromium.launch(headless=False)
    context = browser.new_context()
    page = context.new_page()
    page.goto("https://the-internet.herokuapp.com/login")
    page.get_by_role("textbox", name="Username").click()
    page.get_by_role("textbox", name="Username").click()
    page.get_by_role("textbox", name="Username").fill("tomsmith")
    page.get_by_text("Password", exact=True).click()
    page.get_by_role("textbox", name="Password").fill("SuperSecretPassword!")
    page.get_by_role("button", name=" Login").click()

    # ---------------------
    context.close()
    browser.close()


with sync_playwright() as playwright:
    run(playwright)
🔍 Разбор сгенерированного кода
browser = playwright.chromium.launch(headless=False): Мы просим запустить движок Chromium (база для Chrome).
 Параметр headless=False открывает окно, чтобы мы видели процесс.

context = browser.new_context(): Контекст — это как "инкогнито-вкладка". В нем хранятся свои куки и история.
 Это изолированная среда.

page = context.new_page(): Это сама вкладка в браузере.

page.get_by_role(...): Это те самые Локаторы, о которых мы говорили. Робот решил искать поля ввода по их 
роли (textbox) и названию.

page.fill(...): Команда на ввод текста.

Что в этом коде "не профессионально":
Этот код написан как линейный скрипт. В реальной работе мы используем Pytest. Он позволяет не писать 
вручную запуск браузера, закрытие контекста и прочую "рутину" в каждом тесте.
